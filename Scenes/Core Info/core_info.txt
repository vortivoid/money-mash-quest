### This script stores all the info in the game that should be saved when saving the game, as well as storing values for other scripts to access ###
extends Node

const save_path = "save.data"

func save():
	var savefile = FileAccess.open(save_path, FileAccess.WRITE)
	savefile.store_var(coins)
	savefile.store_var(mouseSteroidsOwned)
	savefile.store_var(autoClickersOwned)
	savefile.store_var(gemsOwned)
	savefile.store_var(planeOwned)
	savefile.store_var(gunOwned)
	savefile.store_string("Why are you here, Huh? Lil cheeky punk, you tryna cheat??? That's kinda cringe bro buddy buster.") #Secret message for when someone opens the save.data file

func loadsave():
	if FileAccess.file_exists(save_path):
		var savefile = FileAccess.open(save_path, FileAccess.READ)
		coins = savefile.get_var(coins)
		mouseSteroidsOwned = savefile.get_var(mouseSteroidsOwned)
		autoClickersOwned = savefile.get_var(autoClickersOwned)
		gemsOwned = savefile.get_var(gemsOwned)
		planeOwned = savefile.get_var(planeOwned)
		gunOwned = savefile.get_var(gunOwned)
